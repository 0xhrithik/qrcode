pragma solidity >=0.4.2;
contract test{

    struct Student {
        string name;
        uint rollno;
        uint hash;
    }

    Student[] public students;

    function _createStudent(string memory   _name, uint _rollno, uint _hash) private {
        students.push(Student(_name,_rollno,_hash));
    }

    function _generateRandomHash(string memory  _str, uint _uint) private pure returns (uint) {
          uint rand = uint(keccak256(abi.encodePacked(_str,_uint)));
          return rand;
    }

    function createStudentPublic(string memory  _name, uint _rollno) public {
        uint randHash = _generateRandomHash(_name,_rollno);
        _createStudent(_name,_rollno,randHash);
    }
    }
}
